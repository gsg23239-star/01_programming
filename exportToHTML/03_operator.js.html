// 논리합 //        => 둘 중에 하나라도 true면 true

var gender= "male";

var result4= loginUser === "admin" || gender === "female"

// 논리곱 (&&)        => 둘 다 true여야 true
var result5= loginUser === "admin" && gender === "female";


// 부정 연산자 (!)

var result6= !true;      // false
var result7= !false;     // true

var result10= !!loginUser;      // true    => 값이 있는지 없는지 판결
